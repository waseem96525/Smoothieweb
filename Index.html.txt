<script>
gsap.registerPlugin(ScrollTrigger);

/* BACKGROUND */
gsap.to("body", {
  background:"linear-gradient(#fff1d6,#ffb385)",
  scrollTrigger:{
    trigger:"body",
    start:"top top",
    end:"bottom bottom",
    scrub:true
  }
});

/* PARALLAX */
gsap.to(".blueberry",{ y:-120, scrub:true });
gsap.to(".kale",{ y:-200, scrub:true });
gsap.to(".ginger",{ y:-260, scrub:true });

/* INGREDIENT FLIGHT */
gsap.to(".ing-blue",{ opacity:1, y:350, scrollTrigger:{trigger:"#hero",start:"top center",end:"center center",scrub:true}});
gsap.to(".ing-kale",{ opacity:1, x:-400, scrollTrigger:{trigger:"#hero",start:"top center",end:"center center",scrub:true}});
gsap.to(".ing-ginger",{ opacity:1, y:-400, scrollTrigger:{trigger:"#hero",start:"top center",end:"center center",scrub:true}});

/* LIQUID FILL */
gsap.to(".liquid-fill",{ y:-260, scrollTrigger:{trigger:"#hero",start:"top center",end:"bottom center",scrub:true}});
gsap.to(".liquid-wave",{ y:-260, scrollTrigger:{trigger:"#hero",start:"top center",end:"bottom center",scrub:true}});

/* WAVE MOTION */
gsap.to(".liquid-wave",{
  attr:{ d:"M0 290 Q50 310 100 290 T200 290 V400 H0 Z"},
  repeat:-1,
  yoyo:true,
  duration:2,
  ease:"sine.inOut"
});

/* SPLASH PHYSICS */
function splash(){
  gsap.fromTo(".splash",
    {opacity:1, scale:.3, y:0},
    {opacity:0, y:-60, scale:1, duration:.6, stagger:.1, ease:"power2.out"}
  );

  gsap.fromTo(".liquid-wave",
    {scaleY:1},
    {scaleY:1.12, duration:.3, repeat:1, yoyo:true, ease:"sine.inOut"}
  );
}

gsap.utils.toArray(".ingredient").forEach(el=>{
  ScrollTrigger.create({
    trigger:el,
    start:"center center",
    onEnter:splash
  });
});

/* TEXT */
gsap.to(".headline span",{
  opacity:1,
  scaleX:1,
  stagger:.15,
  duration:1.2,
  scrollTrigger:{ trigger:".headline", start:"top 80%" }
});
</script>
